CREATE TABLE EMPLOYEE
(
EMP_ID NUMBER(5) NOT NULL ,
E_FULL_NAME VARCHAR2(40) NOT NULL,
E_EMAIL_ID VARCHAR2(40) UNIQUE NOT NULL,
E_MOB_NO NUMBER(10) UNIQUE NOT NULL,
DEPARTMENT VARCHAR2(20) NOT NULL,
PASSWORD VARCHAR2(20) NOT NULL,
E_DATE DATE,
SESSION_ID VARCHAR2(20) NOT NULL,
LANGUAGE CHAR(12) NOT NULL,
PROFILE_PHOTO BLOB,
PRIMARY KEY (EMP_ID)
);
